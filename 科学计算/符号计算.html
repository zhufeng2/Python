证明球体的体积

from sympy import *
#这个positive=true相当重要，正整数
r=symbols('r',positive=true)
x,y=symbols('x,y')
circle=2*integrate(sqrt(r**2-x**2),(x,-r,r))
circle_x=circle.subs(r,sqrt(r**2-x**2))
v=integrate(circle_x,(x,-r,r))
v
